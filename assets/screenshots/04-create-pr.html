# ============================================================================

# 04-create-pr.yml — PR Creator

# ============================================================================

# Creates a pull request from the ghcp-mod branch to main, containing the

# modernization assessment results. Adds labels and a rich PR body.

#

# Repository variables:

#   TARGET_ORG — GitHub organization that owns the target repo

#

# Secrets:

#   GH_PAT — Personal Access Token with repo scope

# ============================================================================



name: "04 · Create PR"



on:

  workflow_dispatch:

    inputs:

      target_repo:

        description: "Repository name to create PR for (just the name)"

        required: true

        type: string



# Only one PR creation per repo at a time

concurrency:

  group: create-pr-${{ inputs.target_repo }}

  cancel-in-progress: false



jobs:

  create-pr:

    name: "Create PR for ${{ inputs.target_repo }}"

    runs-on: ubuntu-latest



    env:

      ORG: ${{ vars.TARGET_ORG }}

      REPO_NAME: ${{ inputs.target_repo }}

      GH_TOKEN: ${{ secrets.GH_PAT }}



    steps:

      # ── Validate inputs ────────────────────────────────────────────────

      - name: Validate inputs

        run: |

          if [[ -z "$ORG" ]]; then

            echo "::error::TARGET_ORG repo variable is not set."

            exit 1

          fi

          if [[ -z "$REPO_NAME" ]]; then

            echo "::error::target_repo input is required."

            exit 1

          fi

          echo "🏢 Organization: $ORG"

          echo "📦 Target repo: $REPO_NAME"



      # ── Check that ghcp-mod branch exists ──────────────────────────────

      - name: Verify ghcp-mod branch exists

        run: |

          echo "🔎 Checking for ghcp-mod branch on $ORG/$REPO_NAME ..."



          BRANCH_EXISTS=$(gh api "repos/$ORG/$REPO_NAME/branches/ghcp-mod" --jq '.name' 2>/dev/null || echo "")

          if [[ -z "$BRANCH_EXISTS" ]]; then

            echo "::error::Branch 'ghcp-mod' does not exist on $ORG/$REPO_NAME. Run the assessment workflow first."

            exit 1

          fi



          echo "✅ Branch ghcp-mod found."



      # ── Check for existing PR ──────────────────────────────────────────

      - name: Check for existing PR

        id: check-pr

        run: |

          EXISTING_PR=$(gh pr list \

            --repo "$ORG/$REPO_NAME" \

            --head ghcp-mod \

            --base main \

            --json number,url \

            --jq '.[0].url // ""')



          if [[ -n "$EXISTING_PR" ]]; then

            echo "ℹ️  PR already exists: $EXISTING_PR"

            echo "exists=true" >> "$GITHUB_OUTPUT"

            echo "url=$EXISTING_PR" >> "$GITHUB_OUTPUT"

          else

            echo "exists=false" >> "$GITHUB_OUTPUT"

          fi



      # ── Create the PR ──────────────────────────────────────────────────

      - name: Create pull request

        id: create

        if: steps.check-pr.outputs.exists == 'false'

        run: |

          PR_TITLE="🔍 GHCP Modernization Assessment — $REPO_NAME"



          PR_BODY=$(cat <<'ENDOFBODY'

          ## 🔍 GHCP Modernization Assessment



          This pull request contains an automated modernization assessment generated by

          the **GitHub Copilot Modernization CLI** (`modernize assess`).



          ### 📂 What's included



          - **Assessment report** in `ghcp-mod-assessment/` — a detailed analysis of the

            codebase covering architecture, dependencies, patterns, and modernization

            opportunities.



          ### 📋 Next steps for the repo owner



          1. **Review the assessment** — Open the files in `ghcp-mod-assessment/` and

             review the findings, recommendations, and identified patterns.

          2. **Create a modernization plan** — Use `modernize plan create` to generate a

             step-by-step modernization plan based on the assessment.

          3. **Prioritize changes** — Identify quick wins vs. larger refactoring efforts.

          4. **Merge or close** — If the assessment is useful, merge this PR to keep the

             report in the repository. Otherwise, close it.



          ### ⚙️ About this PR



          - **Generated by:** GitHub Actions automation

          - **Branch:** `ghcp-mod`

          - **Tool:** [GitHub Copilot Modernization CLI](https://github.com/github/gh-modernize)



          ---



          > 🤖 _This PR was created automatically. Please review the assessment and take

          > action as appropriate for your project._

          ENDOFBODY

          )



          PR_URL=$(gh pr create \

            --repo "$ORG/$REPO_NAME" \

            --title "$PR_TITLE" \

            --body "$PR_BODY" \

            --base main \

            --head ghcp-mod)



          echo "url=$PR_URL" >> "$GITHUB_OUTPUT"

          echo "✅ PR created: $PR_URL"



      # ── Add labels ─────────────────────────────────────────────────────

      - name: Add labels to PR

        if: steps.check-pr.outputs.exists == 'false'

        run: |

          # Ensure labels exist (create if missing)

          for LABEL in "ghcp-mod" "assessment"; do

            gh label create "$LABEL" \

              --repo "$ORG/$REPO_NAME" \

              --description "GHCP Modernization" \

              --color "0E8A16" \

              --force 2>/dev/null || true

          done



          # Add labels to the PR

          PR_URL="${{ steps.create.outputs.url }}"

          PR_NUMBER=$(basename "$PR_URL")



          gh pr edit "$PR_NUMBER" \

            --repo "$ORG/$REPO_NAME" \

            --add-label "ghcp-mod,assessment"



          echo "🏷️ Labels added: ghcp-mod, assessment"



      # ── Job summary ────────────────────────────────────────────────────

      - name: Write job summary

        run: |

          if [[ "${{ steps.check-pr.outputs.exists }}" == "true" ]]; then

            PR_URL="${{ steps.check-pr.outputs.url }}"

            STATUS="ℹ️ PR already existed"

          else

            PR_URL="${{ steps.create.outputs.url }}"

            STATUS="✅ PR created"

          fi



          {

            echo "## 📝 PR Creation — \`$ORG/$REPO_NAME\`"

            echo ""

            echo "| Detail | Value |"

            echo "|--------|-------|"

            echo "| Status | $STATUS |"

            echo "| Repository | [\`$ORG/$REPO_NAME\`](https://github.com/$ORG/$REPO_NAME) |"

            echo "| Pull Request | [$PR_URL]($PR_URL) |"

            echo "| Labels | \`ghcp-mod\`, \`assessment\` |"

            echo ""

            echo "_Run at $(date -u '+%Y-%m-%d %H:%M UTC') by @${{ github.actor }}_"

          } >> "$GITHUB_STEP_SUMMARY"
